1//! Platform initialization.
2
3use lazyinit::LazyInit;
4
5/// boot argument.
6/// This is typically the device tree blob address passed from the bootloader.
7pub static BOOT_ARG: LazyInit<usize> = LazyInit::new();
8
9/// Platform initialization interface.
10#[def_plat_interface]
11pub trait InitIf {
12    /// Initializes the platform at the early stage for the primary core.
13    ///
14    /// This function should be called immediately after the kernel has booted,
15    /// and performed earliest platform configuration and initialization (e.g.,
16    /// early console, clocking).
17    ///
18    /// # Arguments
19    ///
20    /// * `cpu_id` is the logical CPU ID (0, 1, ..., N-1, N is the number of CPU
21    /// cores on the platform).
22    /// * `arg` is passed from the bootloader (typically the device tree blob
23    /// address).
24    ///
25    /// # Before calling this function
26    ///
27    /// * CPU is booted in the kernel mode.
28    /// * Early page table is set up, virtual memory is enabled.
29    /// * CPU-local data is initialized.
30    ///
31    /// # After calling this function
32    ///
33    /// * Exception & interrupt handlers are set up.
34    /// * Early console is initialized.
35    /// * Current monotonic time and wall time can be obtained.
36    fn init_early(cpu_id: usize, arg: usize);
37
38    /// Initializes the platform at the early stage for secondary cores.
39    ///
40    /// See [`init_early`] for details.
41    #[cfg(feature = "smp")]
42    fn init_early_secondary(cpu_id: usize);
43
44    /// Initializes the platform at the later stage for the primary core.
45    ///
46    /// This function should be called after the kernel has done part of its
47    /// initialization (e.g, logging, memory management), and finalized the rest of
48    /// platform configuration and initialization.
49    ///
50    /// # Arguments
51    ///
52    /// * `cpu_id` is the logical CPU ID (0, 1, ..., N-1, N is the number of CPU
53    /// cores on the platform).
54    /// * `arg` is passed from the bootloader (typically the device tree blob
55    /// address).
56    ///
57    /// # Before calling this function
58    ///
59    /// * Kernel logging is initialized.
60    /// * Fine-grained kernel page table is set up (if applicable).
61    /// * Physical memory allocation is initialized (if applicable).
62    ///
63    /// # After calling this function
64    ///
65    /// * Interrupt controller is initialized (if applicable).
66    /// * Timer interrupts are enabled (if applicable).
67    /// * Other essential peripherals are initialized.
68    fn init_later(cpu_id: usize, arg: usize);
69
70    /// Initializes the platform at the later stage for secondary cores.
71    ///
72    /// See [`init_later`] for details.
73    #[cfg(feature = "smp")]
74    fn init_later_secondary(cpu_id: usize);
75}
